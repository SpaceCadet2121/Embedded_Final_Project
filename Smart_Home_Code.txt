static float TMP36_ReadTempC(void)
{
  static float filteredTemp = 25.0f;
  uint32_t sum = 0;
  uint32_t adcValue;
  float voltage;
  float tempC;
  int i;

  for (i = 0; i < 16; i++)
  {
    HAL_ADC_Start(&hadc1);
    HAL_ADC_PollForConversion(&hadc1, HAL_MAX_DELAY);
    sum += HAL_ADC_GetValue(&hadc1);
    HAL_ADC_Stop(&hadc1);

    HAL_Delay(2);
  }

  adcValue = sum / 16;
  voltage = (3.3f * adcValue) / 4095.0f;
  tempC = (voltage - 0.5f) * 100.0f;

  filteredTemp = (0.8f * filteredTemp) + (0.2f * tempC);

  return filteredTemp;
}

static void LCD_Write4Bits(uint8_t nibble)
{
  HAL_GPIO_WritePin(LCD_D4_GPIO_Port, LCD_D4_Pin, (nibble & 0x01) ? GPIO_PIN_SET : GPIO_PIN_RESET);
  HAL_GPIO_WritePin(LCD_D5_GPIO_Port, LCD_D5_Pin, (nibble & 0x02) ? GPIO_PIN_SET : GPIO_PIN_RESET);
  HAL_GPIO_WritePin(LCD_D6_GPIO_Port, LCD_D6_Pin, (nibble & 0x04) ? GPIO_PIN_SET : GPIO_PIN_RESET);
  HAL_GPIO_WritePin(LCD_D7_GPIO_Port, LCD_D7_Pin, (nibble & 0x08) ? GPIO_PIN_SET : GPIO_PIN_RESET);
  LCD_EnablePulse();
}

static void LCD_SendCommand(uint8_t cmd)
{
  HAL_GPIO_WritePin(LCD_RS_GPIO_Port, LCD_RS_Pin, GPIO_PIN_RESET);
  LCD_Write4Bits(cmd >> 4);
  LCD_Write4Bits(cmd & 0x0F);
  HAL_Delay(2);
}

static void LCD_SendData(uint8_t data)
{
  HAL_GPIO_WritePin(LCD_RS_GPIO_Port, LCD_RS_Pin, GPIO_PIN_SET);
  LCD_Write4Bits(data >> 4);
  LCD_Write4Bits(data & 0x0F);
  HAL_Delay(1);
}

static void LCD_Init(void)
{
  HAL_Delay(40);

  HAL_GPIO_WritePin(LCD_RS_GPIO_Port, LCD_RS_Pin, GPIO_PIN_RESET);
  HAL_GPIO_WritePin(LCD_E_GPIO_Port, LCD_E_Pin, GPIO_PIN_RESET);

  LCD_Write4Bits(0x03);
  HAL_Delay(5);
  LCD_Write4Bits(0x03);
  HAL_Delay(5);
  LCD_Write4Bits(0x03);
  HAL_Delay(1);
  LCD_Write4Bits(0x02);

  LCD_SendCommand(0x28);
  LCD_SendCommand(0x0C);
  LCD_SendCommand(0x06);
  LCD_SendCommand(0x01);
  HAL_Delay(2);
}

static void LCD_Clear(void)
{
  LCD_SendCommand(0x01);
  HAL_Delay(2);
}

static void LCD_SetCursor(uint8_t row, uint8_t col)
{
  uint8_t addr = (row == 0) ? (0x80 + col) : (0xC0 + col);
  LCD_SendCommand(addr);
}

static void LCD_SendString(char *str)
{
  while (*str)
  {
    LCD_SendData((uint8_t)(*str));
    str++;
  }
}

/**
  * @brief  Function implementing the SensorTask thread.
  * @param  argument: Not used
  * @retval None
  */
/* USER CODE END Header_StartSensorTask */
void StartSensorTask(void const * argument)
{
  SensorData_t *msg;
  uint8_t currentMotion;

  for(;;)
  {
    g_temperatureC = TMP36_ReadTempC();

    currentMotion = HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_1) ? 1 : 0;
    g_motionState = currentMotion;

    msg = osPoolAlloc(sensorPoolHandle);
    if (msg != NULL)
    {
      msg->temperatureC = g_temperatureC;
      msg->motion = g_motionState;
      msg->systemEnabled = g_systemEnabled;
      msg->lightOn = g_lightOn;
      msg->fanPercent = g_fanPercent;

      if (osMessagePut(SensorQueueHandle, (uint32_t)msg, 0) != osOK)
      {
        osPoolFree(sensorPoolHandle, msg);
      }
    }

    osDelay(200);
  }
}
/* USER CODE BEGIN Header_StartDisplayTask */
/**
* @brief Function implementing the DisplayTask thread.
* @param argument: Not used
* @retval None
*/
/* USER CODE END Header_StartDisplayTask */
void StartDisplayTask(void const * argument)
{
  osEvent evt;
  SensorData_t *msg;
  char line1[17];
  char line2[17];
  float tempF;
  int t10;

  LCD_Init();

  for(;;)
  {
    /* If system OFF → blank LCD */
    if (!g_systemEnabled)
    {
      osMutexWait(lcdMutexHandle, osWaitForever);

      LCD_Clear();
      LCD_SetCursor(0, 0);
      LCD_SendString("                "); // 16 spaces
      LCD_SetCursor(1, 0);
      LCD_SendString("                ");

      osMutexRelease(lcdMutexHandle);

      osDelay(300);
      continue;
    }

    evt = osMessageGet(SensorQueueHandle, osWaitForever);

    if (evt.status == osEventMessage)
    {
      msg = (SensorData_t *)evt.value.p;

      tempF = (msg->temperatureC * 9.0f / 5.0f) + 32.0f;
      t10 = (int)(tempF * 10.0f);

      snprintf(line1, sizeof(line1), "T:%2d.%1dF M:%d",
               t10 / 10, abs(t10 % 10), msg->motion);

      snprintf(line2, sizeof(line2), "L:%d F:%3d%% S:%d",
               msg->lightOn, msg->fanPercent, msg->systemEnabled);

      osMutexWait(lcdMutexHandle, osWaitForever);

      LCD_SetCursor(0, 0);
      LCD_SendString(line1);

      LCD_SetCursor(1, 0);
      LCD_SendString(line2);

      osMutexRelease(lcdMutexHandle);

      osPoolFree(sensorPoolHandle, msg);
    }
  }
}

/* USER CODE BEGIN Header_StartControlTask */
/**
* @brief Function implementing the ControlTask thread.
* @param argument: Not used
* @retval None
*/
/* USER CODE END Header_StartControlTask */
void StartControlTask(void const * argument)
{
  float tempF;
  uint8_t angle;

  for(;;)
  {
    if (!g_systemEnabled)
    {
      Light_Set(0);
      Servo_SetAngle(0);
      g_fanPercent = 0;
      osTimerStop(LightTimerHandle);
      osDelay(200);
      continue;
    }

    /* LED stays on only while motion is currently detected */
    if (g_motionState)
      Light_Set(1);
    else
      Light_Set(0);

    /* Fan control in Fahrenheit */
    tempF = (g_temperatureC * 9.0f / 5.0f) + 32.0f;

    if (tempF < 75.0f)
      angle = 0;
    else if (tempF < 80.0f)
      angle = 60;
    else if (tempF < 85.0f)
      angle = 120;
    else
      angle = 180;

    Servo_SetAngle(angle);

    if (angle == 0) g_fanPercent = 0;
    else if (angle == 60) g_fanPercent = 33;
    else if (angle == 120) g_fanPercent = 66;
    else g_fanPercent = 100;

    osDelay(200);
  }
}
  /* USER CODE END StartControlTask */


/* USER CODE BEGIN Header_StartButtonTask */
/**
* @brief Function implementing the ButtonTask thread.
* @param argument: Not used
* @retval None
*/
/* USER CODE END Header_StartButtonTask */
void StartButtonTask(void const * argument)
{
  for(;;)
  {
    if (osSemaphoreWait(ButtonSemaphoreHandle, osWaitForever) == osOK)
    {
      osDelay(200);   // debounce

      g_systemEnabled = !g_systemEnabled;

      if (!g_systemEnabled)
      {
        /* Turn everything OFF immediately */
        Light_Set(0);
        Servo_SetAngle(0);
        g_fanPercent = 0;
        osTimerStop(LightTimerHandle);
      }
    }
  }
}
